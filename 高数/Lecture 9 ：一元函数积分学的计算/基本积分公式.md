①  
$$  
\begin{cases}  
\int x^kdx=\dfrac{1}{k+1}x^{k+1}+C, k\ne-1;\\[6pt]  
\int \dfrac{1}{x^2}dx=-\dfrac{1}{x}+C;\\[6pt]  
\int \dfrac{1}{\sqrt{x}}dx=2\sqrt{x}+C.  
\end{cases}  
$$

②  
$$  
\int \frac{1}{x}dx=\ln|x|+C.  
$$

③  
$$  
\begin{cases}  
\int e^xdx=e^x+C;\\[6pt]  
\int a^xdx=\dfrac{a^x}{\ln a}+C, a>0,\ a\ne1.  
\end{cases}  
$$

④  
$$  
\begin{cases}  
\int \sin xdx=-\cos x+C;\\[6pt]  
\int \cos xdx=\sin x+C;\\[6pt]  
\int \tan xdx=-\ln|\cos x|+C;\\[6pt]  
\int \cot xdx=\ln|\sin x|+C;\\[6pt]  
\int \dfrac{dx}{\cos x}=\int \sec xdx=\ln|\sec x+\tan x|+C;\\[6pt]  
\int \dfrac{dx}{\sin x}=\int \csc xdx=\ln|\csc x-\cot x|+C;\\[6pt]  
\int \sec^2 xdx=\tan x+C;\\[6pt]  
\int \csc^2 xdx=-\cot x+C;\\[6pt]  
\int \sec x\tan xdx=\sec x+C;\\[6pt]  
\int \csc x\cot xdx=-\csc x+C.  
\end{cases}  
$$

⑤  
$$  
\begin{cases}  
\int \dfrac{1}{1+x^2}dx=\arctan x+C;\\[8pt]  
\int \dfrac{1}{a^2+x^2}dx=\dfrac{1}{a}\arctan\dfrac{x}{a}+C\ (a>0).  
\end{cases}  
$$

⑥  
$$  
\begin{cases}  
\int \dfrac{1}{\sqrt{1-x^2}}dx=\arcsin x+C;\\[8pt]  
\int \dfrac{1}{\sqrt{a^2-x^2}}dx=\arcsin\dfrac{x}{a}+C\ (a>0).  
\end{cases}  
$$

⑦  
$$  
\begin{cases}  
\int \dfrac{1}{\sqrt{x^2+a^2}}dx=\ln\left(x+\sqrt{x^2+a^2}\right)+C\ (\text{常见}a=1);\\[8pt]  
\int \dfrac{1}{\sqrt{x^2-a^2}}dx=\ln\left|x+\sqrt{x^2-a^2}\right|+C\ (|x|>|a|).  
\end{cases}  
$$

⑧  
$$  
\begin{cases}  
\int \dfrac{1}{x^2-a^2}dx=\dfrac{1}{2a}\ln\left|\dfrac{x-a}{x+a}\right|+C;\\[10pt]  
\int \dfrac{1}{a^2-x^2}dx=\dfrac{1}{2a}\ln\left|\dfrac{x+a}{x-a}\right|+C.  
\end{cases}  
$$

⑨  
$$  
\int \sqrt{a^2-x^2}dx=\frac{a^2}{2}\arcsin\frac{x}{a}+\frac{x}{2}\sqrt{a^2-x^2}+C\ (a>|x|\ge0).  
$$

⑩  
$$  
\begin{cases}  
\int \sin^2 xdx=\dfrac{x}{2}-\dfrac{\sin 2x}{4}+C;\\[8pt]  
\int \cos^2 xdx=\dfrac{x}{2}+\dfrac{\sin 2x}{4}+C;\\[8pt]  
\int \tan^2 xdx=\tan x-x+C;\\[8pt]  
\int \cot^2 xdx=-\cot x-x+C.  
\end{cases}  
$$
