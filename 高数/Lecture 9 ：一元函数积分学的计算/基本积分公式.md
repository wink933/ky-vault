①  
$$  
\left{  
\begin{aligned}  
\int x^k,dx&=\frac{1}{k+1}x^{k+1}+C,\ k\ne-1;\  
\int \dfrac{1}{x^2},dx&=-\dfrac{1}{x}+C,\  
\int \dfrac{1}{\sqrt{x}},dx&=2\sqrt{x}+C.  
\end{aligned}  
\right.  
$$

②  
$$  
\int \frac{1}{x},dx=\ln|x|+C.  
$$

③  
$$  
\left{  
\begin{aligned}  
\int e^x,dx&=e^x+C;\  
\int a^x,dx&=\frac{a^x}{\ln a}+C,\ a>0\ \text{且}\ a\ne1.  
\end{aligned}  
\right.  
$$

④  
$$  
\left{  
\begin{aligned}  
\int \sin x,dx&=-\cos x+C;\  
\int \cos x,dx&=\sin x+C;\  
\int \tan x,dx&=-\ln|\cos x|+C;\  
\int \cot x,dx&=\ln|\sin x|+C;\  
\int \frac{dx}{\cos x}=\int \sec x,dx&=\ln|\sec x+\tan x|+C;\  
\int \frac{dx}{\sin x}=\int \csc x,dx&=\ln|\csc x-\cot x|+C;\  
\int \sec^2 x,dx&=\tan x+C;\  
\int \csc^2 x,dx&=-\cot x+C;\  
\int \sec x\tan x,dx&=\sec x+C;\  
\int \csc x\cot x,dx&=-\csc x+C.  
\end{aligned}  
\right.  
$$

⑤  
$$  
\left{  
\begin{aligned}  
\int \dfrac{1}{1+x^2},dx&=\arctan x+C,\  
\int \dfrac{1}{a^2+x^2},dx&=\dfrac{1}{a}\arctan\dfrac{x}{a}+C\ (a>0).  
\end{aligned}  
\right.  
$$

⑥  
$$  
\left{  
\begin{aligned}  
\int \dfrac{1}{\sqrt{1-x^2}},dx&=\arcsin x+C,\  
\int \dfrac{1}{\sqrt{a^2-x^2}},dx&=\arcsin\dfrac{x}{a}+C\ (a>0).  
\end{aligned}  
\right.  
$$

⑦  
$$  
\left{  
\begin{aligned}  
\int \dfrac{1}{\sqrt{x^2+a^2}},dx&=\ln\bigl(x+\sqrt{x^2+a^2}\bigr)+C\ (\text{常见}a=1),\  
\int \dfrac{1}{\sqrt{x^2-a^2}},dx&=\ln\left|x+\sqrt{x^2-a^2}\right|+C\ (|x|>|a|).  
\end{aligned}  
\right.  
$$

⑧  
$$  
\left{  
\begin{aligned}  
\int \frac{1}{x^2-a^2},dx&=\frac{1}{2a}\ln\left|\frac{x-a}{x+a}\right|+C,\  
\int \frac{1}{a^2-x^2},dx&=\frac{1}{2a}\ln\left|\frac{x+a}{x-a}\right|+C.  
\end{aligned}  
\right.  
$$

⑨  
$$  
\int \sqrt{a^2-x^2},dx=\frac{a^2}{2}\arcsin\frac{x}{a}+\frac{x}{2}\sqrt{a^2-x^2}+C\ (a>|x|\ge0).  
$$

⑩  
$$  
\left{  
\begin{aligned}  
\int \sin^2 x,dx&=\frac{x}{2}-\frac{\sin 2x}{4}+C,\  
\int \cos^2 x,dx&=\frac{x}{2}+\frac{\sin 2x}{4}+C,\  
\int \tan^2 x,dx&=\tan x-x+C,\  
\int \cot^2 x,dx&=-\cot x-x+C.  
\end{aligned}  
\right.  
$$
